    const mongoose = require('../config/mongoose');
const Schema = mongoose.Schema;
const config = require('../config');
const init = require('../config/init-app');
const User = require('./User');

/**
 * @class Message
 * @name Models.Message
 * @description модель повідомлення
 * @memberOf Models
 */
const schema = new Schema({
  serviceName: {
    type: String,
    required: true
  },
  status: String,
  groups: Array,
  users: Array,
  addressee: {
    type: String,
    required: true
  },
  title: String,
  icon: String,
  text: {
    type: String,
    required: true
  },
  dispatchTime: {
    type: String,
    default: new Date().toGMTString()
  },
  sendingServer: {
    type: Array,
    required: true
  },
  isViewed: {
    type: Boolean,
    default: false
  },
  createdDate: {
    type: String,
    required: true
  },
  actualityViewedDate: {
    type: String,
    required: true
  },
  actualityNotViewedDate: {
    type: String,
    required: true
  }
});

/**
 * @alias Message.prototype.add
 * @name Models.Message.add
 *
 * @method
 * @description Метод додає повідомлення в БД
 * @param {Object} message - об'єкт повідомлення згідно схеми моделі
 * @param {string} message.serviceName - сервіс, який відправив повідомлення
 * @param {string} message.status - статус повідомлення
 * @param {string} message.title - заголовок повідомлення
 * @param {string} message.icon - uri іконки повідомлення
 * @param {string} message.text - текст повідомлення
 * @param {string} message.dispatchTime - дата відправки повідомлення в GMT String
 * @param {Array} message.sendingServer - сервер відправки повідомлення
 * @param {Boolean} message.isViewed - переглянуто повідомлення чи ні
 * @param {string} message.createdDate - дата створення повідомлення в GMT String
 * @param {string} message.actualityViewedDate - дата коли видалити
 * повідомлення якщо його переглянули в GMT String
 * @param {string} message.actualityNotViewedDate - дата коли видалити
 * повідомлення якщо його НЕ переглянули в GMT String
 *
 * @returns {Promise<object>} - проміс, який містить об'єкт доданого повідомлення
 * згідно схеми моделі, якщо fulfilled
 *
 */
schema.statics.add = async function (message) {
  let users = await User.getUsersByGroups(message.groups);
  let address = message.users.concat(...users).onlyUnique();
  const individualMessages = address.map( (addressee) => {
    return {...message, addressee: addressee}
  });
  return new Promise((resolve, reject) => {
    this.insertMany(individualMessages, (err, docs) => {
      if (err) {
        reject(err);
        if (process.env.NODE_ENV === 'development') throw new Error();
      }
      resolve(docs);
    });
  });
};

    });
  });
};

/**
 * @alias Message.prototype.deleteOldMessages
 * @name Models.Message.deleteOldMessages
 *
 * @method
 * @description Метод видаляє переглянуті та не переглянуті повідомлення в яких сплив
 * термін актуальності, які встановлюються в налаштуваннях системи
 *
 * @returns {Promise} - проміс, який містить масив повідомлень цього користувача,
 */
schema.statics.deleteOldMessages = function () {
  const viewedExpression = 'this.isViewed === true && new Date(this.actualityViewedDate).getTime() < new Date().getTime()';
  const notViewedExpression = 'this.isViewed === false && new Date(this.actualityNotViewedDate).getTime() < new Date().getTime()';

  return new Promise((resolve, reject) => {
    this.deleteMany({$where: viewedExpression + ' && ' + notViewedExpression}, (err) => {
      if (err) {
        reject(err);
        if (process.env.NODE_ENV === 'development') throw new Error();
      }
      resolve();
    });
  });
};

/**
 * @alias Message.prototype.findBySendTime
 * @name Models.Message.findBySendTime
 *
 * @method
 * @description Отримання повідомлень в яких підходить час відправки, тобто
 * вони мають бути відправлені в поточну ітерації петлі
 *
 * @returns {Promise<Array>} - проміс, який містить масив повідомлень
 */
schema.statics.findBySendTime = function () {
  return new Promise((resolve, reject) => {
    const timePeriod = 30 * 60 * 1000 /*config.get('messages:loopSpacing') * 60 * 1000*/;
    const now = new Date().getTime();
    const ex = `${now} <= new Date(this.dispatchTime).getTime() 
      && new Date(this.dispatchTime).getTime() <= (new Date(this.dispatchTime).getTime() + ${timePeriod})`
    this.find({$where: ex},
      (err, docs) => {
        if (err) {
          reject(err);
          if (process.env.NODE_ENV === 'development') throw new Error();
        }
        resolve(docs);
      });
  });
};

/**
 * @alias Message.prototype.messageCreator
 * @name Models.Message.messageCreator
 *
 * @method
 * @description Метод розширює вхідний об'єкт повідомлень, додає сервісні властивості:
 * дата створення, дата закінчення актуальності, тощо.
 *
 * @param {Object} message - об'єкт повідомлення згідно схеми моделі
 * @param {string} message.serviceName - сервіс, який відправив повідомлення
 * @param {string} message.status - статус повідомлення
 * @param {string} message.title - заголовок повідомлення
 * @param {string} message.icon - uri іконки повідомлення
 * @param {string} message.text - текст повідомлення
 * @param {string} message.dispatchTime - дата відправки повідомлення в GMT String
 * @param {Array} message.sendingServer - сервер відправки повідомлення
 * @param {Boolean} message.isViewed - переглянуто повідомлення чи ні
 * @param {string} message.createdDate - дата створення повідомлення в GMT String
 * @param {string} message.actualityViewedDate - дата коли видалити
 * повідомлення якщо його переглянули в GMT String
 * @param {string} message.actualityNotViewedDate - дата коли видалити
 * повідомлення якщо його НЕ переглянули в GMT String
 *
 * @return {object}
 */
schema.statics.messageCreator = function (message) {
  const ACTUALITY_VIEWED = config.get('messages:actualityViewedTime');
  const ACTUALITY_NOT_VIEWED = config.get('messages:actualityNotViewedTime');
  //Константи актуальності прочитаних і не прочитаних повідомлень в годинах
  let now = new Date();

  this.serviceName = message.serviceName;
  this.status = message.status;
  this.groups = Array.isArray(message.groups) ? message.groups : [message.groups];
  this.users = Array.isArray(message.users) ? message.users : [message.users];
  this.title = message.title;
  this.text = message.text;
  this.icon = message.icon;
  this.dispatchTime = message.dispatchTime ? new Date(message.dispatchTime).toGMTString() : now.toGMTString();
  this.sendingServer = Array.isArray(message.sendingServer) ? message.sendingServer : [message.sendingServer];
  this.isViewed = false;
  this.createdDate = now.toGMTString();
  this.actualityViewedDate = new Date().addHours(ACTUALITY_VIEWED).toGMTString();
  this.actualityNotViewedDate = new Date().addHours(ACTUALITY_NOT_VIEWED).toGMTString();
};

const Message = mongoose.model('Message', schema);

module.exports = Message;
